Задача модуля:
Создать документ с SQL-операторами создания необходимых таблиц в базе данных staging, которую мы создадим в следующем модуле.

Решение(SQL код):
create database staging;
use staging;

create table nobel_laureates (
year int, 
category varchar,
prize varchar,
motivation varchar,
prize_share varchar,
laureate_id int,
laureate_type varchar,
full_name varchar,
birth_date date,
birth_city varchar,
birth_country varchar,
sex varchar,
organization_name varchar,
organization_city varchar,
organization_country varchar,
death_date date,
death_city varchar,
death_country varchar );

create table countries_of_the_world (
country varchar,
region varchar,
population int,
area int,
pop_dencity int,
coast_line int,
net_migration int,
infant_mortality int,
gdp int,
literacy int,
phones_per_1000 int,
arable decimal(5,2),
crops decimal(5,2),
other decimal(5,2),
climate int,
birthrate decimal(5,2),
deatrate decimal(5,2)
agriculture decimal(5,2),
industry decimal(5,2),
service decimal(5,2) );

create table continent (
iso2_country_code varchar,
iso2_continent_code varchar );

create table currency (
iso2_country_code varchar,
currency varchar );

create table iso3_country (
iso2_country_code varchar,
iso3_country_code varchar );

create table country_names (
iso2_country_code varchar,
country varchar );

create table phone (
iso2_country_code varchar,
phone int );
